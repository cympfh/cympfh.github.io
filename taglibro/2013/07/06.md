function match(ls, u, f) {

}

function len(ls) {
  return match(ls, 0, function(x,xs){ return 1+len(xs) });

function cons(x,xs) {
  return [x].concat(xs);
}
function filter(ls, pred) {
  return match(ls,[]
    , function(x,xs){
        var ys = filter(xs,pred);
        return pred(x) ? cons(x, ys) : ys;
    });
}
function oddp(x) { return x%2 != 0 }
console.log(filter([1,2,3,4,5], oddp))

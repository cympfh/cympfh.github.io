function mult(n, bi) {
    var base = 19;
    bi = bi.map(function(s){return s*n});
    for (var i=0;i<bi.length;++i) {
        var s = bi[i] + "";
        if (s.length > base) {
            var len = s.length;
            bi[i] = +s.slice(len-base, len);
            if ((i+1) in bi)
                bi[i+1] += +s.slice(0, len-base);
            else
                bi[i+1] = +s.slice(0, len-base);
        }
    }
    return bi;
}
function show(bi) {
    var str = "", len = bi.length;
    for (var i=0;i<len;++i)
        str += bi[len-i-1];
    return str;
}
function fact(n){
    return n==0 ? [1] : mult(n, fact(n-1));

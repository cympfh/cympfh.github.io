#!/bin/bash

TMP=`mktemp`

if [ $# == 0 ]; then
  cat > $TMP
else
  cp $1 $TMP
fi

echo "month      average        yen  days"
for m in {01..12}; do
  SUM=$(
  cat $TMP | grep "^$m" | sort -k1,1 -n | grep '    ' | awk -v s=0 '{s+=$2} END{print s}'
  )
  DAYS=$(
  cat $TMP | grep "^$m" | sort -k1,1 -n | grep '    ' | cut -f1 | uniq | wc -l
  )

  if [ $DAYS -gt 0 ]; then
    echo -n "$m    "
    ruby -e "print sprintf('%12.3f %10d %5d', $SUM/$DAYS.to_f, $SUM, $DAYS)"
    echo
  fi
done

rm $TMP

(use srfi-19 :only (current-date date-week-day))

(define date
  (let1 a (current-date)
  (let ((m (ref a 'month))
        (d (ref a 'day))
        (y (ref a 'year)))
  (let1 dd (mod (- (date-week-day a) 4) 7)
  (let1 d (- d dd)
  (when (<= d 0)
    (dec! m)
    (set! d (+ d (ref '(0 31 29 31 30 31 30 31 31 30 31 30 31) m)))
    (when (<= m 0)
      (dec! y)))
  (format "~a~2,'0D~2,'0D"
          (substring (number->string y) 2 4) m d))))))

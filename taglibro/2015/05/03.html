int solve(string s) {
  vector<int> f(9); rep (i, 9) f[i] = s[i] == 's' ? 0 : s[i] == 'o' ? 1 : 2;
  bool b = num_empty(f) % 2 == 1; // is o's turn?
  return search(f, b);
}

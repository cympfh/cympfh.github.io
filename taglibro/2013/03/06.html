(define-syntax choose
    (syntax-rules ()
        ((_) (fail))
        ((_ a) a)
        ((_ a b ...) (let ((fail0 fail))
            (call/cc (lambda (cc)
                (set! fail (lambda () (set! fail fail0) (cc (choose b ...))))
                (cc a))))) ))
